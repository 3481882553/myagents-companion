### 1.3 失败原因

所有 16 个失败测试都是同一个问题——`@testing-library/react-native` 的 `render` 返回 Promise，测试中没有 `await`：

```typescript
// ❌ 错误：render 返回 Promise，需要 await
it('测试用例', () => {
  const { getByText } = render(<Component />);
  expect(getByText('text')).toBeTruthy();
});

// ✅ 正确：添加 async/await
it('测试用例', async () => {
  const { getByText } = await render(<Component />);
  expect(getByText('text')).toBeTruthy();
});
```

**根因**：`@testing-library/react-native` v12+ 的 `render` 是异步的，但旧测试没有更新。

### 2.2 具体问题

#### 问题 1：渲染测试太浅

```typescript
// 当前：只检查渲染
it('渲染聊天界面', () => {
  const result = ChatScreen({ sessionId: 'ses_001', messages: [], onBack: mockOnBack });
  expect(result).toBeTruthy();  // 只检查不为 null
});

// 应该：检查具体内容
it('渲染聊天界面', async () => {
  const { getByText } = await render(
    <ChatScreen sessionId="ses_001" messages={[]} onBack={mockOnBack} />
  );
  expect(getByText('会话 ses_001')).toBeTruthy();  // 检查标题
  expect(getByText('暂无消息')).toBeTruthy();       // 检查空状态
});
```

#### 问题 2：缺少异步测试

```typescript
// 当前：没有测试 API 调用
it('加载消息', () => {
  const result = ChatScreen({ sessionId: 'ses_001', host: 'localhost:32107' });
  expect(result).toBeTruthy();
});

// 应该：mock fetch，测试加载逻辑
it('加载消息', async () => {
  global.fetch = jest.fn().mockResolvedValue({
    json: () => Promise.resolve({ messages: [{ id: '1', role: 'user', content: 'Hi' }] }),
  });
  
  const { getByText } = await render(
    <ChatScreen sessionId="ses_001" host="localhost:32107" token="test" />
  );
  
  await waitFor(() => {
    expect(getByText('Hi')).toBeTruthy();
  });
});
```

#### 问题 3：缺少边界测试

```typescript
// 当前：没有测试空值

  const result = ToolCallRow({ tool: { name: 'Bash', input: '{"command":"ls"}' } });

it('渲染工具调用 - 空 input', async () => {

it('渲染工具调用 - 超长 input', async () => {
  const longInput = 'x'.repeat(10000);
  const { getByText } = await render(
    <ToolCallRow tool={{ name: 'Bash', input: longInput }} />
  );
  expect(getByText('Bash')).toBeTruthy();
});

it('渲染工具调用 - 无效 JSON', async () => {
  const { getByText } = await render(
    <ToolCallRow tool={{ name: 'Bash', input: 'not-json' }} />
  );
  expect(getByText('Bash')).toBeTruthy();
});
```

## 四、测试规范

### 4.1 命名规范

```typescript
describe('组件名称', () => {
  describe('功能模块', () => {
    it('应该在什么条件下做什么', async () => {
      // Arrange: 准备测试数据
      // Act: 执行操作
      // Assert: 验证结果
    });
  });
});
```

### 4.2 Mock 规范

```typescript
// ✅ 正确：Mock 外部依赖
jest.mock('react-native', () => ({
  StyleSheet: { create: (s: any) => s },
  View: 'View',
  Text: 'Text',
}));

// ✅ 正确：Mock API 调用
global.fetch = jest.fn().mockResolvedValue({
  json: () => Promise.resolve({ data: 'test' }),
});

// ❌ 错误：Mock 内部实现
jest.mock('../utils/helper', () => ({
  helper: jest.fn(),  // 不要 mock 内部函数
}));
```

### 4.3 断言规范

```typescript
// ✅ 正确：具体断言
expect(getByText('标题')).toBeTruthy();
expect(mockFn).toHaveBeenCalledWith('参数');
expect(result).toHaveLength(3);

// ❌ 错误：模糊断言
expect(result).toBeTruthy();  // 太模糊
expect(result).toBeDefined(); // 太模糊
```
