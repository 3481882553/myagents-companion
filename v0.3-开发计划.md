#### D3-D4：前缀固化流式渲染策略（6h）

**目标**：实现 OpenOmniBot 启发的 prefix-freeze 策略——已完成部分用稳定 Markdown 渲染，新到来的尾部用纯文本淡入，定期 flush 重解析。

**核心算法**：

```typescript
// src/components/markdown/StreamingMessageRenderer.tsx (新文件)

interface StreamingState {
  fullText: string;              // 完整文本
  renderedPrefixLength: number;  // 已固化的前缀长度
  isStreaming: boolean;          // 是否正在流式传输
}

function StreamingMessageRenderer({ text, isStreaming }: Props) {
  const [state, setState] = useState<StreamingState>({
    fullText: text,
    renderedPrefixLength: isStreaming ? lastSafeBoundary(text) : text.length,
    isStreaming,
  });
  
  // 流式结束时，重新解析全部文本
  useEffect(() => {
    if (!isStreaming) {
      setState({ fullText: text, renderedPrefixLength: text.length, isStreaming: false });
    }
  }, [isStreaming, text]);
  
  // 流式进行中：定期 flush（每 500ms 或每 500 字符）
  useEffect(() => {
    if (!isStreaming) return;
    const timer = setInterval(() => {
      setState(prev => {
        const newLength = lastSafeBoundary(prev.fullText);
        if (newLength > prev.renderedPrefixLength) {
          return { ...prev, renderedPrefixLength: newLength };
        }
        return prev;
      });
    }, 500);
    return () => clearInterval(timer);
  }, [isStreaming, text]);
  
  const stablePrefix = state.fullText.slice(0, state.renderedPrefixLength);
  const streamingTail = state.fullText.slice(state.renderedPrefixLength);
  
  return (
    <View>
      {/* 前缀：稳定 Markdown 渲染 */}
      <MarkdownRenderer content={stablePrefix} />
      {/* 尾部：纯文本 + 淡入动画 */}
      {streamingTail.length > 0 && (
        <Animated.Text style={{ opacity: fadeAnim }}>
          {streamingTail}
        </Animated.Text>
      )}
    </View>
  );
}

/**
 * 找到最后一个安全的截断边界：
 * 1. 完整的段落（双换行后）
 * 2. 完整的代码块（``` 闭合）
 * 3. 完整的表格行
 * 4. 完整的列表项
 */
function lastSafeBoundary(text: string): number {
  // 优先找完整段落
  const lastParagraph = text.lastIndexOf('\n\n');
  if (lastParagraph > text.length * 0.3) return lastParagraph + 2;
  
  // 检查代码块是否闭合
  const codeBlockStarts = (text.match(/```/g) || []).length;
  if (codeBlockStarts % 2 === 0) {
    // 代码块完整，在最后一个闭合的 ``` 后截断
    const lastClose = text.lastIndexOf('```\n');
    if (lastClose > 0) return lastClose + 4;
  }
  
  // 找完整行（最近的换行）
  const lastNewline = text.lastIndexOf('\n');
  if (lastNewline > 0) return lastNewline + 1;
  
  return text.length;
}
```

```typescript
interface ProcessRowProps {
  icon: string;           // 工具图标（emoji 或 Lucide 组件）
  name: string;           // 工具名（Bash / Read / Grep）
  subtitle?: string;      // 副标签（命令摘要 / 文件路径）
  state: 'running' | 'completed' | 'error' | 'interrupted';
  isThinking?: boolean;   // 是否为思考过程
  onPress: () => void;    // 点击展开/收起
  expanded: boolean;
}

function ProcessRow({ icon, name, subtitle, state, isThinking, onPress, expanded }: ProcessRowProps) {
  return (
    <TouchableOpacity onPress={onPress} style={processRowStyles[state]}>
      <View style={styles.row}>
        {/* 状态点 */}
        <View style={[styles.dot, dotColors[state]]} />
        {/* 图标 */}
        <Text style={styles.icon}>{icon}</Text>
        {/* 工具名 */}
        <Text style={styles.name}>{name}</Text>
        {/* 副标签 */}
        {subtitle && <Text style={styles.subtitle} numberOfLines={1}>{subtitle}</Text>}
        {/* 展开箭头 */}
        <Text style={styles.arrow}>{expanded ? '▼' : '▶'}</Text>
      </View>
      {/* 展开详情 */}
      {expanded && <ToolUseDetail name={name} />}
    </TouchableOpacity>
  );
}
```

  onEndReached={() => loadOlderMessages()}
  onEndReachedThreshold={0.5}
/>
```

## 十、v0.4 展望：Widget 渲染 + 离线缓存（3 周）

### 10.1 为什么 Widget 提前到 v0.4

| 原计划 | 新计划 | 原因 |

##### D1：WidgetRenderer 核心组件（4h）

**目标**：实现 `WidgetRenderer.tsx`，与桌面端安全模型对齐。

```
桌面端                              移动端
<iframe sandbox="allow-scripts">  → <WebView>
  无 allow-same-origin               originWhitelist=['about:blank']
  CSP default-src 'none'             javaScriptEnabled=true（仅 Widget HTML）
  postMessage                   →    onMessage / injectJavaScript
  sanitizeForStreaming()        →    同样逻辑，注入前执行
```

**组件状态机**：

```
loading → WebView 初始化
   ├── onMessage('widget:ready') → ready（显示内容）
   └── 5s 超时 → error（重试 1 次）
         └── 连续 3 次失败 → fallback（降级为代码文本）
```

##### D2：sanitize 安全过滤（2h）

**目标**：实现与桌面端一致的 sanitize 逻辑。

```typescript
// 移植桌面端 sanitizeForStreaming() 逻辑
function sanitizeWidgetHtml(html: string): string {
  return html
    .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
    .replace(/\son\w+\s*=/gi, ' data-removed-')
    .replace(/javascript:/gi, 'void:');
}
```

**交付物**：`src/utils/sanitize.ts` + 安全测试用例（XSS 攻击向量）

##### D3：Widget 通信协议（2h）

| 方向 | 事件 | 数据 | 说明 |

| WebView → RN | `widget:error` | `{ error }` | 运行时错误 → 累加失败计数 |
