import { HelperSessionService } from '../HelperSessionService';
import { MockStorage, MockSidecarApi } from '../../../__tests__/mocks';

describe('HelperSessionService', () => {
  let service: HelperSessionService;
  let storage: MockStorage;
  let api: MockSidecarApi;

  beforeEach(() => {
    storage = new MockStorage();
    api = new MockSidecarApi();
    service = new HelperSessionService(storage, api);
  });

  describe('ensureHelperSession', () => {
    it('返回已缓存的 sessionId', async () => {
      storage.set('helper.sessionId', 'existing-id');
      const result = await service.ensureHelperSession();
      expect(result).toBe('existing-id');
      expect(api.post).not.toHaveBeenCalled();
    });

    it('创建新会话并缓存 sessionId', async () => {
      api.post.mockResolvedValue({ sessionId: 'new-id' });
      const result = await service.ensureHelperSession();
      expect(result).toBe('new-id');
      expect(storage.getString('helper.sessionId')).toBe('new-id');
    });

    it('优先从桌面端同步最新 CLAUDE.md', async () => {
      api.get.mockResolvedValue({ version: '999', claudeMd: 'latest', supportSkill: 'latest' });
      api.post.mockResolvedValue({ sessionId: 'id' });
      await service.ensureHelperSession();
      expect(api.post).toHaveBeenCalledWith('/api/session/create-helper', expect.objectContaining({
        claudeMd: 'latest',
      }));
    });

    it('Sidecar 不支持 bundled-files 端点时使用 App 内置版本', async () => {
      api.get.mockRejectedValue(new Error('Not Found'));
      api.post.mockResolvedValue({ sessionId: 'id' });
      await service.ensureHelperSession();
      expect(api.post).toHaveBeenCalledWith('/api/session/create-helper', expect.objectContaining({
        claudeMd: expect.stringContaining('你是 MyAgents'),  // 内置 CLAUDE.md 开头

    });
  });

        expect.stringContaining('/support')

      api.get.mockResolvedValue([]);

import { SseClient, EventPriority } from '../sse-client';
import { SSE_EVENTS_FIXTURE } from '../../__tests__/mocks/sse-events';

describe('SseClient', () => {
  let client: SseClient;

      token: 'test-token',
    });
  });

  describe('连接状态机', () => {
    it('初始状态为 DISCONNECTED', () => {
      expect(client.state).toBe('DISCONNECTED');
    });

    it('connect() 转换到 CONNECTING', () => {
      client.connect();
      expect(client.state).toBe('CONNECTING');
    });

    it('连接成功转换到 CONNECTED', () => {
      client.connect();
      client.simulateOpen(); // 测试辅助方法
      expect(client.state).toBe('CONNECTED');
    });

    it('连接失败转换到 RETRYING', () => {
      client.connect();
      client.simulateError();
      expect(client.state).toBe('RETRYING');
    });
  });

  describe('事件优先级', () => {
    it('critical 事件立即分发', () => {
      const handler = jest.fn();
      client.on('chat:message-complete', handler);
      client.simulateMessage({ type: 'chat:message-complete', data: {} });
      expect(handler).toHaveBeenCalledTimes(1);
    });

    it('coalescible 事件在 40ms 窗口内合并', async () => {
      const handler = jest.fn();
      client.on('chat:message-chunk', handler);
      client.simulateMessage({ type: 'chat:message-chunk', data: { text: 'Hello' } });
      client.simulateMessage({ type: 'chat:message-chunk', data: { text: ' World' } });
      expect(handler).not.toHaveBeenCalled();
      await new Promise(r => setTimeout(r, 50));
      expect(handler).toHaveBeenCalledTimes(1);
      expect(handler).toHaveBeenCalledWith(expect.objectContaining({
        data: { text: 'Hello World' },
      }));
    });

    it('droppable 事件不分发', () => {
      const handler = jest.fn();
      client.on('chat:log', handler);
      client.simulateMessage({ type: 'chat:log', data: 'test' });
      expect(handler).not.toHaveBeenCalled();
    });
  });

  describe('重连', () => {
    it('指数退避：500ms → 1s → 2s → ...', async () => {
      client.connect();
      client.simulateError();
      expect(client.retryDelay).toBe(500);
      client.simulateError();
      expect(client.retryDelay).toBe(1000);
      client.simulateError();
      expect(client.retryDelay).toBe(2000);
    });

    it('超过最大重试次数转为 DISCONNECTED', () => {
      client.connect();
      for (let i = 0; i < 11; i++) {
        client.simulateError();
      }
      expect(client.state).toBe('DISCONNECTED');
    });
  });

  describe('JSON 解析', () => {
    it('JSON_EVENTS 中的事件解析为对象', () => {
      const handler = jest.fn();
      client.on('chat:init', handler);
      client.simulateMessage({ type: 'chat:init', data: '{"sessionId":"test"}' });
      expect(handler).toHaveBeenCalledWith(expect.objectContaining({
        data: { sessionId: 'test' },
      }));
    });

    it('STRING_EVENTS 中的事件保持字符串', () => {
      const handler = jest.fn();
      client.on('chat:message-chunk', handler);
      client.simulateMessage({ type: 'chat:message-chunk', data: 'Hello' });
      // coalescible，等 40ms
      setTimeout(() => {
        expect(handler).toHaveBeenCalledWith(expect.objectContaining({
          data: 'Hello',
        }));
      }, 50);
    });
  });
});

import { JoseTokenService } from '../token-service';

describe('JoseTokenService', () => {
  let service: JoseTokenService;

  beforeEach(() => {
    const secret = new TextEncoder().encode('test-secret-key-at-least-32-chars!');
    service = new JoseTokenService(secret);
  });

  describe('sign', () => {
    it('签发有效 JWT', async () => {
      const token = await service.sign({ paired: true }, '7d');
      expect(token).toBeTruthy();
      expect(token.split('.')).toHaveLength(3); // header.payload.signature
    });
  });

  describe('verify', () => {
    it('验证有效 Token', async () => {
      const token = await service.sign({ paired: true }, '7d');
      const payload = await service.verify(token);
      expect(payload.paired).toBe(true);
    });

    it('拒绝过期 Token', async () => {
      const token = await service.sign({ paired: true }, '0s'); // 立即过期
      await new Promise(r => setTimeout(r, 100));
      await expect(service.verify(token)).rejects.toThrow();
    });

    it('拒绝篡改的 Token', async () => {
      const token = await service.sign({ paired: true }, '7d');
      const tampered = token.slice(0, -5) + 'XXXXX';
      await expect(service.verify(tampered)).rejects.toThrow();
    });

    it('拒绝不同 secret 签发的 Token', async () => {
      const otherService = new JoseTokenService(
        new TextEncoder().encode('different-secret-key-at-least-32!')
      );
      const token = await otherService.sign({ paired: true }, '7d');
      await expect(service.verify(token)).rejects.toThrow();
    });
  });
});
